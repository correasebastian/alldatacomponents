@mixin cols-direct-child-not-col {
  > *:not([class*=" col"]):not([class^="col"]):not(.field__items) {
    @content;
  }
}

@mixin cols-drupal-direct-child-not-col {
  .field__items > *:not([class*=" col"]):not([class^="col"]) {
    @content;
  }
}

    @extend .col;
